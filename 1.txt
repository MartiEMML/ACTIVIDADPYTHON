# Programa que pide tres números enteros y muestra cuál es el mayor

# Ingresar tres números
num1 = int(input("Ingrese el primer número: "))
num2 = int(input("Ingrese el segundo número: "))
num3 = int(input("Ingrese el tercer número: "))

# Comparar para encontrar el mayor
mayor = num1  # asumimos que el primero es el mayor

if num2 > mayor:
    mayor = num2
if num3 > mayor:
    mayor = num3

# Mostrar resultado
print(f"\nEl número mayor es: {mayor}")
